167  { 
168      const char t[10] = " "; 
169      char *token; 
170      char tmp; 
171      int flag=1; 
172   
173      /* get the first token */ 
174      token = strtok(s, t); 
175      tmp =token[0]; 
176   
177      /* walk through other tokens */ 
178      while (token != NULL) { 
179  //        printf(" %s\n", token); 
180  //        printf(" %c\n",token[0]);
